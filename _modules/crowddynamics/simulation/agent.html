[docs]def shape_validator(*dimensions):
    """Validate the shape of :class:`traittypes.Array` trait.

    Args:
        *dimensions (int): Dimensions of the array
    
    Returns:
        Callable[trait, value]: Validator function
    """

    def validator(trait, value):
        if value.shape != dimensions:
            raise TraitError(
                'Expected an of shape %s and got and array with shape %s' % (
                    dimensions, value.shape))
        else:
            return value

    return validator




[docs]def length_validator(*lengths):
    """Validates the length of :class:`traittypes.Array` trait.

    Args:
        *lengths (numbers.Real): 
        
    Returns:
        Callable[trait, value]: Validator function
    """
    def validator(trait, value):
        l = length(value)
        if np.allclose(l, lengths):
            raise TraitError(
                'Expected an of length %s and got and array with shape %s' % (
                    lengths, l))
        else:
            return value

    return validator




[docs]def trait_to_dtype(name, trait):
    """Convert TraitType to numpy.dtype in format (name, dtype, shape)::
    
        Int -> numpy.int64
        Float -> numpy.float64
        Complex -> numpy.complex128
        Bool -> numpy.bool_

    Args:
        name (str): 
        trait (TraitType): 
    
    Returns:
        (str, numpy.dtype, tuple):
    """
    # TODO: other types
    if isinstance(trait, TraitType):
        if isinstance(trait, Int):
            return name, np.int64
        elif isinstance(trait, Float):
            return name, np.float64
        elif isinstance(trait, Complex):
            return name, np.complex128
        elif isinstance(trait, Bool):
            return name, np.bool_
        elif isinstance(trait, Array):
            return name, trait.dtype, trait.default_value.shape
        else:
            raise NotImplementedError(
                'Trait conversion is not implemented for: {}'.format(trait))
    else:
        raise TypeError('Trait should be instance of {}'.format(TraitType))




[docs]def class_traits(cls):
    """Traverse the class hierarchy and accumulate all traits.

    Args:
        cls: Subclass of HasTraits

    Yields:
        (name, TraitType): Tuple of trait name and type
    """
    for c in inspect.getmro(cls):
        if issubclass(c, HasTraits):
            for key, value in vars(c).items():
                if isinstance(value, TraitType):
                    yield key, value

        """Structured numpy.dtype for forming an array of the value of agent
        type.

        Returns:
            numpy.dtype: Numpy structured dtype for the agent type
        """
        fields = [trait_to_dtype(k, v) for k, v in class_traits(cls)]
        return np.dtype(fields)


    def __array__(self):
        """Array interface for using ``numpy.array`` on the agent type. 
        
        Returns:
            numpy.ndarray:
        """
        dtype = self.dtype()
        values = tuple(getattr(self, name) for name in dtype.fields)
        return np.array([values], dtype=dtype)

    def _default_configs(self):

    @observe('configs_path')
    def _observe_configs_path(self, change):
        path = change['new']

    @staticmethod
    def _truncnorm(mean, abs_scale):
        """Individual value from truncnorm"""
        return np.asscalar(truncnorm(-3.0, 3.0, loc=mean, abs_scale=abs_scale))

    @default('inertia_rot')
    def _default_inertia_rot(self):
        """Rotational inertia of mass 80 kg and radius 0.27 m agent. Should be
        scaled to correct value for agents."""
        # TODO: raise error is mass and radius are not defined
        inertia = 4.0 * np.pi
        mass = 80.0
        radius = 0.27
        return inertia * (self.mass / mass) * (self.radius / radius) ** 2

    @default('radius')
    def _default_radius(self):
        return self._truncnorm(self.radius_mean, self.radius_scale)

    @default('target_velocity')
    def _default_target_velocity(self):
        return self._truncnorm(self.velocity_mean, self.velocity_scale)

    @default('mass')
    def _default_mass(self):
        return self._truncnorm(self.mass_mean, self.mass_scale)

    @default('r_t')
    def _default_r_t(self):
        return self.r_t * self.radius

    @default('r_s')
    def _default_r_s(self):
        return self.r_s * self.radius

    @default('r_ts')
    def _default_r_ts(self):
        return self.r_ts * self.radius

[docs]    def add_group(self, size, attributes, check_overlapping=True):
        """Add group of agents

        Args:
            size (int):
            attributes:
            check_overlapping (bool):
        """
        array = np.zeros(size, dtype=self.type.dtype())
        index = 0
        overlaps = 0
        overlaps_max = 10 * size

        while index < size and overlaps < overlaps_max:
            new_attributes = self.next(attributes)
            # TODO: body_types
            new_agent = self.type(new_attributes)

            if check_overlapping and index:
                neighbours = self._neighbours.nearest(new_agent.position,
                                                      radius=1.0)
                if new_agent.overlapping(array[neighbours]):
                    # Agent is overlapping
                    overlaps += 1
                    continue

            # Agent can be successfully placed
            array[index][:] = np.array(new_agent)
            self._neighbours[new_agent.position] = index
            index += 1

        self.array = np.concatenate((self.array, array))
