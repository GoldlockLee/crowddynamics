@numba.jit(i8[:](i8, i8[:], i8[:]), nopython=True, nogil=True, cache=True)

[docs]def split_by_interactions(parts, neigh, count):
    """Split blocklist equally by number of interactions for multithreading"""
    interactions_tot = 0
    interactions = np.zeros_like(count)
    for i, count_i in enumerate(count):
        interactions_tot += (count_i - 1) ** 2
        for j in neigh:
            k = i + j
            if 0 <= k < len(count):
                interactions_tot += count_i * count[k]
        interactions[i] = interactions_tot

    splits = np.zeros(parts+1, dtype=np.int64)
    splits[-1] = len(count)
    num = 1
    size = np.int64(interactions_tot / parts)
    s = 0
    for i, interaction in enumerate(interactions):
        s += interaction
        if s >= num * size:
            splits[num] = i
            num += 1

    return splits




[docs]def flat_index(indices, shape):
    i = indices[0]
    for j in range(1, len(indices)):
        i *= shape[j]
        i += indices[j]
    return i




[docs]def agent_agent_block_list_multithreaded(agents):
    max_agent_radius = 0.3
    index_list, count, offset, shape = block_list(
        agents['position'], SIGTH_SOC + 2 * max_agent_radius)
    indices = np.arange(len(agents))
    neigh = np.array([flat_index(i, shape) for i in
                      ((1, 0), (1, 1), (0, 1), (1, -1))])

    # Split by interactions
    THREADS = 3
    splits = split_by_interactions(THREADS, neigh, count)
    chunks = [(agents, indices[splits[i]:splits[i+1]], neigh, index_list, count,
               offset) for i in range(THREADS)]

    if is_model(agents, 'circular'):
        target = agent_agent_block_list_circular
    elif is_model(agents, 'three_circle'):
        target = agent_agent_block_list_three_circle
    else:
        raise Exception

    threads = [threading.Thread(target=target, args=chunk) for chunk in chunks]
    for thread in threads:
        thread.start()
    for thread in threads:
        thread.join()
