    def f():
        keys = ((random.uniform(-1.0, 1.0) // cell_size,
                 random.uniform(-1.0, 1.0) // cell_size) for _ in range(size))
        values = range(size)
        d = defaultdict(list)
        for key, value in zip(keys, values):
            d[key].append(value)

    benchmark(f)
    assert True
